# WC26 Frontend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the WC26 frontend from the current shadcn dashboard scaffold into a typed, testable World Cup 2026 performance dashboard with mock-backed API contracts, five primary views, and live update behavior.

**Architecture:** Implement a frontend domain layer first, then build screens from canonical WC26 DTOs rather than provider-shaped Sportmonks or API-Football payloads. Use deterministic mock data and App Router route handlers before real backend integration, so every component can be developed and tested without provider credentials.

**Tech Stack:** Next.js 16 App Router, React 19, TypeScript 5.9, Tailwind CSS 4, shadcn/ui, Recharts, TanStack Table, TanStack Virtual, Vitest, React Testing Library, native EventSource/SSE.

Key decisions from research:

- DARKO-style inspiration becomes player leaderboard, comparison, trajectory, scatterplot-ready data fields, projection freshness, and CSV export.
- Provider APIs must be normalized into canonical DTOs before React components consume data.
- The 2026 third-place qualification race is a primary frontend feature, not a bracket afterthought.
- Live views need connection status, freshness labels, reconnect states, and stale data warnings.
- Physical and prediction features must expose model/source limitations and avoid health or biometric inference.

Create `frontend/lib/domain/standings.ts`:

```ts
import { GROUPS, type GroupCode, type GroupStandingRow, type Match, type QualificationState, type Team } from "./wc26"

interface StandingsInput {
  teams: Team[]
  matches: Match[]
  group: GroupCode
}

interface ThirdPlaceInput {
  teams: Team[]
  matches: Match[]
}

export function getQualificationState(position: number, scope: "group" | "third-place"): QualificationState {
  if (scope === "third-place") {
    return position <= 8 ? "qualified" : "eliminated"
  }

  if (position <= 2) return "qualified"
  if (position === 3) return "third-place-race"
  return "eliminated"
}

export function calculateGroupStandings({ teams, matches, group }: StandingsInput): GroupStandingRow[] {
  const groupTeams = teams.filter((team) => team.group === group)

  const rows = groupTeams.map((team) => {
    const completedMatches = matches.filter(
      (match) =>
        match.group === group &&
        match.status === "full-time" &&
        (match.homeTeamId === team.id || match.awayTeamId === team.id),
    )

    const record = completedMatches.reduce(
      (acc, match) => {
        const isHome = match.homeTeamId === team.id
        const goalsFor = isHome ? match.homeScore : match.awayScore
        const goalsAgainst = isHome ? match.awayScore : match.homeScore

        acc.goalsFor += goalsFor
        acc.goalsAgainst += goalsAgainst

        if (goalsFor > goalsAgainst) {
          acc.wins += 1
          acc.form.push("W")
        } else if (goalsFor === goalsAgainst) {
          acc.draws += 1
          acc.form.push("D")
        } else {
          acc.losses += 1
          acc.form.push("L")
        }

        return acc
      },
      { wins: 0, draws: 0, losses: 0, goalsFor: 0, goalsAgainst: 0, form: [] as Array<"W" | "D" | "L"> },
    )

    return {
      team,
      played: completedMatches.length,
      wins: record.wins,
      draws: record.draws,
      losses: record.losses,
      goalsFor: record.goalsFor,
      goalsAgainst: record.goalsAgainst,
      goalDifference: record.goalsFor - record.goalsAgainst,
      points: record.wins * 3 + record.draws,
      form: record.form,
      qualificationState: "projected" as QualificationState,
    }
  })

  return rows
    .toSorted(compareStandingRows)
    .map((row, index) => ({ ...row, qualificationState: getQualificationState(index + 1, "group") }))
}

export function calculateThirdPlaceRace({ teams, matches }: ThirdPlaceInput): GroupStandingRow[] {
  return GROUPS.map((group) => calculateGroupStandings({ teams, matches, group })[2])
    .toSorted(compareStandingRows)
    .map((row, index) => ({ ...row, qualificationState: getQualificationState(index + 1, "third-place") }))
}

function compareStandingRows(a: GroupStandingRow, b: GroupStandingRow) {
  return (
    b.points - a.points ||
    b.goalDifference - a.goalDifference ||
    b.goalsFor - a.goalsFor ||
    b.team.conductScore - a.team.conductScore ||
    b.team.eloRating - a.team.eloRating
  )
}
```

```ts
type PredictionTotalInput = {
  homeWin: number
  draw: number
  awayWin: number
}

export function formatProbability(value: number) {
  return `${Math.round(value * 100)}%`
}

export function getConfidenceLabel(confidence: number) {
  if (confidence >= 0.75) return "High confidence"
  if (confidence >= 0.55) return "Medium confidence"
  return "Low confidence"
}

export function validatePredictionTotal(prediction: PredictionTotalInput) {
  const total = prediction.homeWin + prediction.draw + prediction.awayWin
  return Math.abs(total - 1) <= 0.01
}
```

```ts
export function createEnvelope<T>(data: T, warnings: string[] = []): ApiEnvelope<T> {
  return {
    data,
    meta: {
      generatedAt: new Date("2026-06-15T12:00:00.000Z").toISOString(),
      source: "mock",
      freshness: "static",
      warnings,
    },
  }
}

export function getTournamentSnapshot() {
  return createEnvelope({
    currentPhase: "group",
    teams: mockTeams,
    players: mockPlayers,
    liveMatch: mockMatches.find((match) => match.status === "live") ?? mockMatches[0],
    groupStandings: Object.fromEntries(
      [...new Set(mockTeams.map((team) => team.group))].map((group) => [
        group,
        calculateGroupStandings({ teams: mockTeams, matches: mockMatches, group }),
      ]),
    ),
    thirdPlaceRace: calculateThirdPlaceRace({ teams: mockTeams, matches: mockMatches }),
  })
}

export function getMatchDetail(matchId: string): ApiEnvelope<Match | null> {
  const match = mockMatches.find((item) => item.id === matchId) ?? null
  return createEnvelope(match, match ? [] : [`Match ${matchId} was not found`])
}
```

```tsx
"use client"

import * as React from "react"
import type { Player, Team } from "@/lib/domain/wc26"

const modes = ["Goals", "xG", "Assists", "Saves"] as const

interface Props {
  players: Player[]
  teams: Team[]
}

export function TopPerformersRail({ players, teams }: Props) {
  const [mode, setMode] = React.useState<(typeof modes)[number]>("Goals")
  const rows = players
    .toSorted((a, b) => getStat(b, mode) - getStat(a, mode))
    .slice(0, 5)

  return (
    <section className="rounded-xl border bg-card p-4" aria-label="Top performers">
      <div className="flex flex-wrap items-center justify-between gap-3">
        <h2 className="text-title-md">Top performers</h2>
        <div role="tablist" aria-label="Performer metric" className="flex gap-1 rounded-full bg-muted p-1">
          {modes.map((item) => (
            <button
              key={item}
              type="button"
              role="tab"
              aria-selected={mode === item}
              className="rounded-full px-3 py-1 text-sm data-[active=true]:bg-background"
              data-active={mode === item}
              onClick={() => setMode(item)}
            >
              {item}
            </button>
          ))}
        </div>
      </div>
      <div className="mt-4 grid gap-3">
        {rows.map((player, index) => {
          const team = teams.find((item) => item.id === player.teamId)

          return (
            <div key={player.id} className="flex items-center justify-between rounded-lg bg-muted px-3 py-2">
              <span>{index + 1}. {player.name} · {team?.fifaCode}</span>
              <span className="font-mono">{getStat(player, mode).toFixed(mode === "xG" ? 2 : 0)}</span>
            </div>
          )
        })}
      </div>
    </section>
  )
}

function getStat(player: Player, mode: (typeof modes)[number]) {
  if (mode === "Goals") return player.stats.goals
  if (mode === "xG") return player.stats.xg
  if (mode === "Assists") return player.stats.assists
  return player.stats.saves
}
```

```ts
"use client"

import * as React from "react"
import type { MatchEvent } from "@/lib/domain/wc26"

type StreamStatus = "connecting" | "open" | "reconnecting" | "closed"

export function useMatchEventStream(matchId: string) {
  const [status, setStatus] = React.useState<StreamStatus>("connecting")
  const [latestEvent, setLatestEvent] = React.useState<MatchEvent | null>(null)
  const [lastHeartbeatAt, setLastHeartbeatAt] = React.useState<string | null>(null)

  React.useEffect(() => {
    if (typeof EventSource === "undefined") {
      setStatus("closed")
      return
    }

    const source = new EventSource(`/api/live/${matchId}`)

    source.onopen = () => setStatus("open")
    source.onerror = () => setStatus("reconnecting")
    source.addEventListener("heartbeat", () => setLastHeartbeatAt(new Date().toISOString()))
    source.addEventListener("match-event", (event) => {
      setLatestEvent(JSON.parse(event.data) as MatchEvent)
    })

    return () => {
      source.close()
      setStatus("closed")
    }
  }, [matchId])

  return { status, latestEvent, lastHeartbeatAt }
}
```
